package com.plantuml.ubrex.builder;

import com.plantuml.ubrex.CompositeList;

public class UBrexConcat extends UBrexPart {

	private UBrexConcat(CompositeList list) {
		super(list);
	}

	public static UBrexConcat build(UBrexPart... all) {
		final CompositeList result = CompositeList.createEmpty();
		for (UBrexPart element : all)
			result.addChallenge(element.getChallenge());

		return new UBrexConcat(result);

	}

}
