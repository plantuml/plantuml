	private static int[] cacheDistSymbol() {
		int[] r = new int[32768];
		for (int i = 0; i < 32768; ++i) {
			r[i] = distSymbol(i);
		}
		return r;
	}

	static int distExtraBitsValue(int dist) {
		if (dist < 5) {
			return 0;
		} else if (dist < 9) {
			return (dist - 5) & 1;
		} else if (dist < 17) {
			return (dist - 9) & 3;
		} else if (dist < 33) {
			return (dist - 17) & 7;
		} else if (dist < 65) {
			return (dist - 33) & 15;
		} else if (dist < 129) {
			return (dist - 65) & 31;
		} else if (dist < 257) {
			return (dist - 129) & 63;
		} else if (dist < 513) {
			return (dist - 257) & 127;
		} else if (dist < 1025) {
			return (dist - 513) & 255;
		} else if (dist < 2049) {
			return (dist - 1025) & 511;
		} else if (dist < 4097) {
			return (dist - 2049) & 1023;
		} else if (dist < 8193) {
			return (dist - 4097) & 2047;
		} else if (dist < 16385) {
			return (dist - 8193) & 4095;
		}
		return (dist - 16385) & 8191;
	}
