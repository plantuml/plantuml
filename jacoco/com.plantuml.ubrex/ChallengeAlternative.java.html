package com.plantuml.ubrex;

import java.util.ArrayList;
import java.util.Collection;

public class ChallengeAlternative implements Challenge {

	private final Collection<Challenge> alternatives = new ArrayList<>();

	@Override
	public String toString() {
		return "ALTERNATIVE" + alternatives;
	}

	@Override
	public ChallengeResult runChallenge(TextNavigator string, int position) {
		for (Challenge pattern : alternatives) {
			final ChallengeResult tmp = pattern.runChallenge(string, position);
			if (tmp.getFullCaptureLength() >= 0)
				return tmp;

		}
		return new ChallengeResult(NO_MATCH);
	}

	public void addAlternative(Challenge pattern) {
		alternatives.add(pattern);
	}

}
