	@DuplicateCode(reference = "SvgGraphics#format")
	private String format(double x) {
		if (x == 0.0)
			return "0";

		String s = String.format(Locale.US, "%.4f", x);

		final int dot = s.indexOf('.');
		if (dot >= 0) {
			int end = s.length() - 1;
			while (end > dot && s.charAt(end) == '0')
				end--;

			if (end == dot)
				end--;

			s = s.substring(0, end + 1);
		}
		return s;
	}

}
