package net.sourceforge.plantuml.skin;

public enum PragmaKey {
	
	// Temporary pragma
	SVGNEWDATA,
	SEQUENCE_MESSAGE_SPAN,
	
	// For debug
	EMULATE_NO_GRAPHVIZ_INSTALLATION,
	EMULATE_GRAPHVIZ_CRASH,
	EMULATE_GRAPHVIZ_244_ON_WINDOWS,

	ASPECT, //
	COMPACT, //
	DEFAULT_LABEL_ANGLE, //
	DEFAULT_LABEL_DISTANCE, //
	EDGE_CORNER_RADIUS, //
	GRAPH_ATTRIBUTES, //
	HORIZONTAL_LINE_BETWEEN_DIFFERENT_PACKAGE_ALLOWED, //
	KERMOR, //
	LABEL_ANGLE, //
	LABEL_DISTANCE, //
	RATIO, //
	SHOW_DEPRECATION, //
	SVG_FONT, //
	SVG_INTERACTIVE, //
	SVEK_TRACE("true"), //
	TEOZ("true"), //
	TEX_SYSTEM, //
	TEX_PREAMBLE, //
	USE_INTERMEDIATE_PACKAGES, //
	USE_VERTICAL_IF;
	
	private final String defaultValue;
	
	private PragmaKey() {
		this(null);
	}

	private PragmaKey(String defaultValue) {
		this.defaultValue = defaultValue;
	}

	public final String getDefaultValue() {
		return defaultValue;
	}


	private static String simplify(String s) {
		final StringBuilder result = new StringBuilder();

		for (char c : s.toCharArray())
			if ((c >= 'A' && c <= 'Z') || (c >= 'a' && c <= 'z'))
				result.append(Character.toLowerCase(c));

		return result.toString();
	}

	public static PragmaKey lazyFrom(String s) {
		for (PragmaKey key : values())
			if (simplify(s).equals(simplify(key.name())))
				return key;
		return null;
	}

}
